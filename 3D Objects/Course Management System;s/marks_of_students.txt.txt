public class Management {
    public static void main(String[] args) {
        createWindow();
    }

    private static void createWindow() {
        JFrame frame = new JFrame("Swing Tester");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        createUI(frame);
        frame.setSize(560, 200);
        frame.setLocationRelativeTo(null);
        frame.setVisible(true);
    }

    private static void createUI(final JFrame frame){
        JPanel panel = new JPanel();
        LayoutManager layout = new FlowLayout();
        panel.setLayout(layout);

        JButton button_1 = new JButton("Student");
        JButton button_2 = new JButton("Teacher");
        JButton button_3 = new JButton("Course Administrator");
        final JLabel label = new JLabel();

        //Optional Button of Student,Teacher,Administrator
        button_1.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                String Student_Id = (String)JOptionPane.showInputDialog(
                        frame,
                        "Enter the Student ID",
                        "Swing Tester",
                        JOptionPane.PLAIN_MESSAGE,
                        null,
                        null,
                        "Red"


                );

                if(Student_Id != null && Student_Id.length() > 0) {
                    label.setText("You selected:" + Student_Id);


                } else {
                    label.setText("None selected");
                }

            }


        });

        button_2.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                String result = (String)JOptionPane.showInputDialog(
                        frame,
                        "Select one of the color",
                        "Swing Tester",
                        JOptionPane.PLAIN_MESSAGE,
                        null,
                        null,
                        "Red"
                );
                if(result != null && result.length() > 0){
                    label.setText("You selected:" + result);
                }else {
                    label.setText("None selected");
                }
            }
        });
        button_3.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                String result = (String)JOptionPane.showInputDialog(
                        frame,
                        "Select one of the color",
                        "Swing Tester",
                        JOptionPane.PLAIN_MESSAGE,
                        null,
                        null,
                        "Red"
                );
                if(result != null && result.length() > 0){
                    label.setText("You selected:" + result);
                }else {
                    label.setText("None selected");
                }
            }
        });
        panel.add(button_1);
        panel.add(button_2);
        panel.add(button_3);

        panel.add(label);

        frame.getContentPane().add(panel, BorderLayout.CENTER);
    }
